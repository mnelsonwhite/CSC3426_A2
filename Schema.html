Water Hockey DB Schema

Pools Table

CREATE TABLE IF NOT EXISTS pools (
    Name TEXT PRIMARY KEY,
    Length INTEGER,
    Address TEXT);

Teams Table

CREATE TABLE IF NOT EXISTS teams (
    Name TEXT PRIMARY KEY,
    PoolName TEXT,
    Manager TEXT,
    FOREIGN KEY (PoolName) REFERENCES pools(Name) ON DELETE CASCADE);

Games Table

CREATE TABLE IF NOT EXISTS games (
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    TeamAName TEXT,
    TeamBName TEXT,
    PoolName TEXT,
    ScoreA INTEGER,
    ScoreB INTEGER,
    Date TEXT,
    FOREIGN KEY (TeamAName) REFERENCES teams(Name) ON DELETE CASCADE,
    FOREIGN KEY (TeamBName) REFERENCES teams(Name) ON DELETE CASCADE,
    FOREIGN KEY (PoolName) REFERENCES pools(Name) ON DELETE CASCADE);

Players Table

CREATE TABLE IF NOT EXISTS players (
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    TeamName TEXT,
    GivenName TEXT,
    FamilyName TEXT,
    Dob TEXT,
    Handed TEXT,
    FOREIGN KEY (TeamName) REFERENCES teams(Name) ON DELETE CASCADE);
